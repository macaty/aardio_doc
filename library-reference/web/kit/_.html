web.kit 成员列表

精简版 WebKit 内核浏览器组件
NPAPI插件加载目录:
~\plugins\ 以及 %localappdata%\aardio\std
p-plugin\plugins
支持在注册表路径 HKCU\Software\MozillaPlugins 注册的 NPAPI 插件,
内核源码: https://github.com/aardio/wke

web.kit.configure(setting)

web.kit.configure({  
    proxy = {  
        type = "HTTP";  
        hostname = "127.0.0.1";  
        port = 8080;  
        username = "";  
        password ="";  
    };  
    cookieFilePath = "/cookie";  
})

web.kit.export(JS变量名,aardio对象)

web.kit.jsBindFunction("函数名",参数个数,函数对象)

web.kit.jsBindFunction("JS函数名",/*参数个数*/,function(){  
    var wkeWebView = owner;  
    var winform = wkeWebView.getForm();  
    winform.close();  
})
