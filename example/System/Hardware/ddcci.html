aardio 范例: 显示器 DDC/CI

//显示器 DDC/CI
//显示器控制接口，不是所有显示器都支持
import console;  
import sys.ddcci;

for ddcci in sys.ddcci.each() {
    console.dump(ddcci.capabilities,ddcci.description);
    if(!ddcci.capabilities.vcp) continue;

    ddcci.setPowerMode(4); //关屏
    sleep(1000);
    ddcci.setPowerMode(1); //亮屏
}

console.pause(true);
