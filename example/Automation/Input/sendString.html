aardio 范例: 发送字符串

//发送字符串
import key;

//发送按键,不区分双字符键,受输入法影响
key.send( "abcdeHIJK12345~!@#$%^&*()_+" ); 

//发送中英文字符串，不受输入法影响
key.sendString("发送字符串abcd")

key.pressEx( "CTRL","A" )
